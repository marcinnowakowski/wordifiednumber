========================================================
|                   Wordified Number                   |
--------------------------------------------------------
|   Installation  and run instruction step by step     |
========================================================
git clone git://github.com/marcinnowakowski/wordifiednumber.git
cd wordifiednumber

# Compile and run tests for the project
mvn test-compile
mvn test

# Import project to eclipse
mvn eclipse:eclipse
# EclipseIDE: 
# File->Import...==>General/Existing Projects into Workspace
# ==>Select root directory: (path to wordifiednumber)->Project: (wordifiednumber)==>Finish
